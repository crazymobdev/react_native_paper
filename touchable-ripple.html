<TouchableRipple />

Ripple provides components with a material "ink ripple" interaction effect

Usage:

const MyComponent = () => (
  <TouchableRipple>
    onPress={() => {}}
    borderless
    rippleColor="rgba(0, 0, 0, .32)"
  >
    <View>
      <Paragraph>Press me</Paragrpah>
    </View>
  </TouchableRipple>
);

Props

borderless: boolean
background: Object
onPress: Function
rippleColor: string
underlayColor: string
*children: unknown
style: any
